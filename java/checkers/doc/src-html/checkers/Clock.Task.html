001    package checkers;
002    
003    import java.util.Timer;
004    import java.util.TimerTask;
005    
006    public class Clock
007    {
008            protected Timer timer;
009            
010            public Clock(Checkers checkers)
011            {
012                    //checkers.timerTick();
013                    timer = new Timer("Clock");
014                    timer.schedule(new Task(checkers), 0, 1000);
015            }
016            
017            public void finalize()
018            {
019                    stop();
020            }
021            
022            public void stop()
023            {
024                    timer.cancel();
025            }
026            
027            public static interface Observer
028            {
029                    public void timerTick();
030            }
031            
032            protected static class Task extends TimerTask
033            {
034                    protected Observer observer;
035                    
036                    public Task(Checkers checkers)
037                    {
038                            observer = (Observer) checkers;
039                    }
040                    
041                    public void run()
042                    {
043                            observer.timerTick();
044                    }
045            }
046    }

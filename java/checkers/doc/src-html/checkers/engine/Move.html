001    package checkers.engine;
002    
003    import java.util.Vector;
004    
005    public class Move extends Vector<Integer> /*implements Comparable<Move>*/
006    {
007            public Integer value;
008            
009            public Move(int... coords)
010            {
011                    for (int coord : coords)
012                            add(coord);
013            }
014            
015            public Move(Move m)
016            {
017                    for (int coord : m)
018                            add(coord);
019            }
020    
021            public Move clone()
022            {
023                    Move newmove = new Move();
024                    for (int c : this)
025                            newmove.add(c);
026                    return newmove;
027            }
028            
029            public boolean equals(Object other)
030            {
031                    if (other.getClass() != Move.class)
032                            return false;
033                    
034                    Move m = (Move) other;
035    
036                    if (m.size() != size())
037                            return false;
038                    
039                    for (int i = 0; i < size(); i++)
040                            if (m.get(i) != get(i))
041                                    return false;
042    
043                    return true;
044            }
045            
046            /*public int compareTo(Move other)

051                            : 0;
052            }*/
053            
054            public String toString()
055            {
056                    if (size() == 2)
057                    {
058                            return String.format("%d-%d", get(0) + 1, get(1) + 1);
059                    }
060                    else
061                    {
062                            String result = "";
063                            for (int i = 0; i < size(); i += 2)
064                            {
065                                    result += get(i) + 1;
066                                    if (i < (size() - 1))
067                                            result += "x";
068                            }
069                            return result;
070                    }
071            }
072    
073            public int applyMine(int mine)
074            {
075                    int b = 1 << get(0);
076                    if ((mine & b) != 0)
077                    {
078                            mine ^= b;
079                            mine ^= 1 << lastElement();
080                    }
081                    return mine;
082            }
083    
084            public int applyHis(int his)
085            {
086                    if (size() > 2)
087                            for (int i = 1; i < size(); i += 2)
088                            {
089                                    int b = 1 << get(i);
090                                    if ((his & b) != 0)
091                                            his ^= 1 << get(i);
092                            }
093                    return his;
094            }
095            
096            public int applyKings(int kings, int crownrow)
097            {
098                    kings = applyMine(applyHis(kings));
099    
100                    int b = 1 << lastElement();
101                    if ((crownrow & b) != 0)
102                            return kings | b;
103    
104                    return kings;
105            }
106    }

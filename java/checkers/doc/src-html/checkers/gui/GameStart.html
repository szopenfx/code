001    package checkers.gui;
002    
003    import java.awt.Button;
004    import java.awt.Checkbox;
005    import java.awt.Color;
006    import java.awt.Component;
007    import java.awt.Frame;
008    import java.awt.GridLayout;
009    import java.awt.Label;
010    import java.awt.TextField;
011    import java.awt.event.ActionEvent;
012    import java.awt.event.ActionListener;
013    import java.awt.event.WindowAdapter;
014    import java.awt.event.WindowEvent;
015    import java.io.BufferedReader;
016    import java.io.FileNotFoundException;
017    import java.io.FileReader;
018    import java.io.IOException;
019    import java.io.PrintWriter;
020    
021    import checkers.Checkers;
022    
023    public class GameStart extends Frame
024    {
025            public TextField name1 = new TextField();
026            public TextField name2 = new TextField();
027            
028            public TextField time1 = new TextField();
029            public TextField time2 = new TextField();
030            
031            public TextField moves1 = new TextField();
032            public TextField moves2 = new TextField();
033            
034            public Checkbox relaxed = new Checkbox("Ignore clock");
035            
036            public Button start = new Button("Go");
037            
038            public static String FILENAME = "checkers.cfg";
039            
040            public GameStart()
041            {
042                    initFrame();
043                    initEvents();
044    
045                    try
046                    {
047                            load();
048                    }
049                    catch (IOException e)
050                    {
051                            defaults();
052                    }
053            }
054            
055            protected void addLabel(String text)
056            {
057                    add(new Label(text));
058            }
059            
060            protected void initFrame()
061            {
062                    setLayout(new GridLayout(8, 3, 5, 5));
063            
064                    addLabel("");
065                    addLabel("Player 1");
066                    addLabel("Player 2");
067                    
068                    addLabel("Name");
069                    add(name1);
070                    add(name2);
071                    
072                    addLabel("");
073                    addLabel("");
074                    addLabel("");
075                    
076                    addLabel("");
077                    addLabel("Moves");
078                    addLabel("Time");
079    
080                    addLabel("Stage 1");
081                    add(moves1);
082                    add(time1);
083                    
084                    addLabel("Stage 2");
085                    add(moves2);
086                    add(time2);
087                    
088                    addLabel("");
089                    addLabel("");
090                    addLabel("");
091                    
092                    add(relaxed);;
093                    addLabel("");
094                    add(start);
095                    
096                    for (Component c : this.getComponents())
097                            c.setBackground(Color.BLACK);
098            
099                    setBackground(Color.BLACK);
100                    setForeground(Color.GREEN);
101    
102                    pack();
103            
104                    setLocationByPlatform(true);
105                    setResizable(false);
106                    setTitle("Checkers");
107                    setVisible(true);
108            }
109    
110            protected void initEvents()
111            {
112                    addWindowListener(new WindowAdapter() {
113                            public void windowClosing(WindowEvent arg0)
114                            {
115                                    System.exit(0);
116                            }
117                    });
118                    
119                    start.addActionListener(new ActionListener() {
120                            public void actionPerformed(ActionEvent e)
121                            {
122                                    try
123                                    {
124                                            save();
125                                    }
126                                    catch (FileNotFoundException ex)
127                                    {
128                                            ex.printStackTrace();
129                                    }
130                                    setVisible(false);
131                                    new Checkers(
132                                                    name1.getText(), 
133                                                    name2.getText(), 
134                                                    60 * Integer.parseInt(time1.getText()), 
135                                                    60 * Integer.parseInt(time2.getText()),
136                                                    Integer.parseInt(moves1.getText()),
137                                                    Integer.parseInt(moves2.getText()),
138                                                    relaxed.getState()
139                                    );
140                            }
141                    });
142            }
143            
144            public void save() throws FileNotFoundException
145            {
146                    PrintWriter pw = new PrintWriter(FILENAME);
147                    
148                    pw.println(name1.getText());
149                    pw.println(name2.getText());
150                    
151                    pw.println(time1.getText());
152                    pw.println(time2.getText());
153                    
154                    pw.println(moves1.getText());
155                    pw.println(moves2.getText());
156                    
157                    pw.println(relaxed.getState() ? "true" : "Who-hah! Got you all in check");
158                    
159                    pw.close();
160            }
161            
162            public void load() throws IOException
163            {
164                    BufferedReader br = new BufferedReader(new FileReader(FILENAME));
165                            
166                    name1.setText(br.readLine());
167                    name2.setText(br.readLine());
168                    
169                    time1.setText(br.readLine());
170                    time2.setText(br.readLine());
171                    
172                    moves1.setText(br.readLine());
173                    moves2.setText(br.readLine());
174                    
175                    relaxed.setState(br.readLine().equals("true"));
176                    
177                    br.close();
178            }
179            
180            public void defaults()
181            {
182                    name1.setText("Human");
183                    name2.setText("Computer");
184                    
185                    time1.setText("5");
186                    time2.setText("5");
187                    
188                    moves1.setText("60");
189                    moves2.setText("60");
190                    
191                    relaxed.setState(true);
192            }
193            
194    }

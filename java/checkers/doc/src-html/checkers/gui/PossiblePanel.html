001    package checkers.gui;
002    
003    import java.awt.Color;
004    import java.awt.Dimension;
005    import java.awt.Font;
006    import java.awt.FontMetrics;
007    import java.awt.Graphics;
008    import java.awt.Rectangle;
009    
010    import checkers.Checkers;
011    import checkers.engine.MoveList;
012    
013    public class PossiblePanel extends DoubleBuffer
014    {
015            protected int top = 0;
016            protected int lineheight;
017            protected int linecount;
018            
019            protected Rectangle buttonup = new Rectangle();
020            protected Rectangle buttondown = new Rectangle();
021            protected Rectangle buttonundo = new Rectangle();
022            
023            protected static final Color TEXTCOLOR = Color.WHITE;
024            protected static final Color BUTTONCOLOR = Color.GRAY;
025            protected static final Color BUTTONTEXT = Color.WHITE;
026    
027            protected MoveList moves = null;
028            
029            public PossiblePanel(Checkers c)
030            {
031                    moves = c.possible;
032            }
033    
034            public void paintBuffer(Graphics g)
035            {
036                    FontMetrics metrics; 
037                    Dimension size = getSize();
038                    
039                    g.setFont(new Font("Courier", Font.PLAIN, 12));
040                    g.setColor(TEXTCOLOR);
041                    
042                    metrics = g.getFontMetrics();
043                    lineheight = metrics.getHeight();
044                    
045                    linecount = size.height / lineheight - 1;
046                    
047                    for (int i = 0; i < linecount - 1; i++)
048                            if ((top + i) < moves.size())
049                            {
050                                    String s = moves.get(top + i).toString();
051                                    g.drawString(s, 0, lineheight + i * lineheight);
052                            }
053                    
054                    buttonup.setBounds(0 * size.width / 3, linecount * lineheight, size.width / 3, lineheight);
055                    buttondown.setBounds(1 * size.width / 3, linecount * lineheight, size.width / 3, lineheight);
056                    //buttonundo.setBounds(2 * size.width / 3, linecount * lineheight, size.width / 3, lineheight);
057                    
058                    g.setFont(new Font("Arial", Font.PLAIN, 12));
059                    metrics = g.getFontMetrics();
060                    lineheight = metrics.getHeight();
061                    
062                    g.setColor(BUTTONCOLOR);
063                    g.drawRect(buttonup.x,   buttonup.y,   buttonup.width,   buttonup.height);
064                    g.drawRect(buttondown.x, buttondown.y, buttondown.width, buttondown.height);
065                    //g.drawRect(buttonundo.x, buttonundo.y, buttonundo.width, buttonundo.height);
066                    
067                    g.setColor(BUTTONTEXT);
068                    g.drawString("up",   buttonup.x   + (buttonup.width   / 2) - (metrics.stringWidth("up"  ) / 2), buttonup.y   + lineheight - 2);
069                    g.drawString("down", buttondown.x + (buttondown.width / 2) - (metrics.stringWidth("down") / 2), buttondown.y + lineheight - 2);
070                    //g.drawString("undo", buttonundo.x + (buttonundo.width / 2) - (metrics.stringWidth("undo") / 2), buttonundo.y + lineheight - 2);
071    
072            }
073    }

001    package checkers.gui;
002    
003    import java.awt.Button;
004    import java.awt.Color;
005    import java.awt.Font;
006    import java.awt.List;
007    import java.awt.Panel;
008    import java.awt.event.ActionEvent;
009    import java.awt.event.ActionListener;
010    
011    import checkers.Checkers;
012    import checkers.engine.Move;
013    
014    // TODO : movepanel pimpen
015    // TODO : forward move panel maken
016    
017    public class MovePanel extends Panel
018    {
019            protected List list = new List();
020            protected Button undo = new Button("Undo");
021            protected Observer observer;
022            
023            public MovePanel(Checkers checkers)
024            {
025                    observer = checkers;
026                    
027                    initPanel();
028            }
029            
030            public static interface Observer
031            {
032                    public void undoMove();
033            }
034    
035            protected void initPanel()
036            {
037                    setLayout(null);
038                    
039                    add(list);
040                    add(undo);
041                    
042                    list.setFont(new Font("Courier", Font.PLAIN, 12));
043                    list.setBounds(0, 0, 200, 486);
044                    list.setBackground(Color.BLACK);
045                    list.setForeground(Color.WHITE);
046                    
047                    undo.setBounds(150, 487, 50, 25);
048                    undo.addActionListener(new ActionListener() {
049                            public void actionPerformed(ActionEvent e)
050                            {
051                                    observer.undoMove();
052                            }
053                    });
054                    undo.setBackground(Color.BLACK);
055                    undo.setForeground(Color.GREEN);
056            }
057            
058            public void addMove(Move m)
059            {
060                    int count = list.getItemCount();
061                    int idx = count / 2;
062                    boolean isp1 = count % 2 == 0;
063    
064                    String text = isp1 
065                                            ? String.format("%2d. %s", idx + 1, m.toString())
066                                            : String.format("    %s", m.toString());
067                    
068                    list.add(text);
069                    list.makeVisible(count);
070            }
071            
072            public void removeMove()
073            {
074                    list.remove(list.getItemCount() - 1);
075                    list.makeVisible(list.getItemCount() - 1);
076            }
077    }

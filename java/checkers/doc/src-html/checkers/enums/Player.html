001    package checkers.enums;
002    
003    import java.awt.Color;
004    
005    import checkers.engine.Board;
006    import checkers.engine.Move;
007    
008    public enum Player
009    { 
010            PLAYER1(0xf0000000, new Direction[] { Direction.NE, Direction.NW, Direction.SE, Direction.SW }, Color.RED, "Player1")
011            {
012                    public int mine(Board b) { return b.player1; }
013                    
014                    public int his(Board b) { return b.player2; }
015                    
016                    public boolean canMove(Direction d) { return d == Direction.NE || d == Direction.NW; }
017                    
018                    public void applyMove(Board b, Move m)
019                    {
020                            b.player1 = m.applyMine(b.player1);
021                            b.player2 = m.applyHis(b.player2);
022                            b.kings = m.applyKings(b.kings, crownrow);
023                    }
024            },
025            
026            PLAYER2(0x0000000f, new Direction[] { Direction.SW, Direction.SE, Direction.NW, Direction.NE }, Color.WHITE, "Player2")
027            {
028                    public int mine(Board b) { return b.player2; }
029                    
030                    public int his(Board b) { return b.player1; }
031                    
032                    public boolean canMove(Direction d) { return d == Direction.SE || d == Direction.SW; }
033                    
034                    public void applyMove(Board b, Move m)
035                    {
036                            b.player1 = m.applyHis(b.player1);
037                            b.player2 = m.applyMine(b.player2);
038                            b.kings = m.applyKings(b.kings, crownrow);;
039                    }
040            };
041            
042            static {
043                    PLAYER1.other = PLAYER2;
044                    PLAYER2.other = PLAYER1;
045            }
046            
047            public int crownrow;
048            public Color color;
049            public String name;
050            public int clock;
051            public int moves;
052            public boolean cpu = false;
053            public Player other;
054            public Phase phase = Phase.PHASE1;
055            public Direction[] directions;
056            
057            Player(int cr, Direction[] d, Color c, String n)
058            {
059                    crownrow = cr;
060                    directions = d;
061                    color = c;
062                    name = n;
063            }
064            
065            public abstract int mine(Board b);
066            public abstract int his(Board b);
067            public abstract boolean canMove(Direction d);
068            public abstract void applyMove(Board b, Move m);
069    };
